const _ = require('lodash');
const Hapi = require('hapi');
const Disinfect = require('disinfect');
const JWT = require('hapi-auth-jwt2');
const Async = require('async');
const Args = require('commander');
const HD = require('@docbrasil/tools');
 
/**
 * Class to start the application
 * @class
 */
class StartApp {
    constructor() {
        const self = this;
        self.server = null;
        self.serverConfig = {};
    }
 
    initArgs(next) {
        const self = this;
 
        // Set Args
        Args
            .version('0.0.1')
            .option('-s, --servertype [type]', 'Type of the server location: dev, test, accept or prod')
            .parse(process.argv);
 
        process.env["servertype"] = Args.servertype ? Args.servertype : "dev";
        next();
    }
    getServerConfig(next) {
        const self = this;
        self.serverConfig = HD.serverconfig();
        next();
    }
    initLoaders(next) {
        const self = this;
        HD.init({
            domains: {
                session: 'session',
                cache: 'helpers',
                redis: 'helpers',
                mongo: 'helpers',
                crudmongo: 'helpers',
                utils: 'helpers',
                config: 'config',
            }
        });
        next();
    }
    initCache(next) {
        const self = this;
        HD.cache().init({
            connection: {
                host: self.serverConfig.redis.host,
                port: self.serverConfig.redis.port
            },
            list: self.serverConfig.cache
        });
        next();
    }
    initMongo(next) {
        const self = this;
        let mongo = HD.mongo();
        mongo.setConfig({ host: self.serverConfig.mongo.host, options: self.serverConfig.mongo.options });
        mongo.connect((err) => {
            next(err);
        });
    }
    createServer(next) {
        const self = this;
        self.server = new Hapi.Server(self.serverConfig.hapi);
        self.server.connection({ port: self.serverConfig.server.port, host: self.serverConfig.server.host });
        next();
    }
    initDisinfect(next) {
        const self = this;
        self.server.register({
            register: Disinfect,
            options: {
                disinfectQuery: true,
                disinfectParams: true,
                disinfectPayload: true
            }
        }, (err) => {
            next(err);
        });
    }
    initJWT(next) {
        const self = this;
        self.server.register({ register: JWT}, (err) => {
            if(!err) {
                let jwtOptions = self.serverConfig.jwt;
                jwtOptions.validateFunc = HD.session().validate;
                self.server.auth.strategy('jwt', 'jwt', true, jwtOptions);
            }
            next(err);
        });
    }
    initRoutes(next) {
        const self = this;
        let finalRoutes = [];
        _.each(self.routes, function(routeType){
            let reqRout = require("./routes/" + routeType);
            finalRoutes = _.union(finalRoutes, reqRout);
        });
        _.each(finalRoutes, (route) => {
            self.server.route(route);
        });
        next();
    }
    startServer(next) {
        const self = this;
        self.server.start((err) => {
            next(err);
        });
    }
    start() {
        const self = this;
        let funcs = [];
 
        funcs.push((next) => self.initArgs(next));
        funcs.push((next) => self.initLoaders(next));
        funcs.push((next) => self.getServerConfig(next));
        funcs.push((next) => self.initCache(next));
        funcs.push((next) => self.initMongo(next));
        funcs.push((next) => self.createServer(next));
        funcs.push((next) => self.initDisinfect(next));
        funcs.push((next) => self.initJWT(next));
        funcs.push((next) => self.initRoutes(next));
        funcs.push((next) => self.startServer(next));
 
        Async.series(funcs, (err) => {
            if (err) {
                console.dir(err);
            } else {
                const dt = new Date();
                console.log(`Server started with URI ${self.server.info.uri} at ${dt.toTimeString()} with options: -s: ${process.env["servertype"]}`);
            }
        });
    }
}
 
const app = new StartApp();
app.start();
