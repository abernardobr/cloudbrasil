/**
 * CRUDInterface: Functions to manipulate all CRUD interfaces
 *
 * Variable naming convension: As you use this service on a class, in order to be accessed from outside and
 * understood by all how to access, there is a naming convension:
 *      Use the name of the class all lower caps.
 *
 *      For example: CRUDInterface --> crudinterface
 *
 *      That is:
 *
 *      const CRUDInterface = require('tofs-webtools').CRUDInterface;
 *      self.CRUDInterface = new CRUDInterface({
 *           timer: true,
 *           callbacks: CRUDCallbacks
 *       });
 *
 *
 *  Defining Callbacks
 *      You have to create a new class with all of the callbacks that you are using. It will be mixed with CRUDInterface.
 *
 *      Example:
 *
 *      class CRUDCallbacks {
 *           findValidate(params, cb) {
 *               cb(null, params);
 *           }
 *
 *           findBefore(params, cb) {
 *               cb(null, params);
 *           }
 *
 *           findAfter(params, cb) {
 *               cb(null, params);
 *           }

 *              cb(null, params);
