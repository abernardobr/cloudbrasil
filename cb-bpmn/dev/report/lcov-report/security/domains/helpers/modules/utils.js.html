class TOTimer {
    constructor(options) {
        const self = this;
        self.options = _.assign({
            precision: 3
        }, options);
        self.marks = [];
        self.start = 0;
    }
 
    init() {
        const self = this;
        self.start = process.hrtime();
    }
 
    mark(tag) {
        const self = this;
        const timer = process.hrtime(self.start);
        const elapsed = timer[1] / 1000000; // divide by a million to get nano to milli
        self.marks.push({ tag: tag, time: `${timer[0]} s,  ${elapsed.toFixed(self.options.precision)} ms` });
    }
 
    log(message) {
        const self = this;
        let sOut = '';
        _.each(self.marks, (mark) => {
            sOut += `[ tag: ${mark.tag}  -  ${mark.time} ]`;
        });
        console.log(message ? (`${message} - ${sOut}`) : sOut);
    }
}
