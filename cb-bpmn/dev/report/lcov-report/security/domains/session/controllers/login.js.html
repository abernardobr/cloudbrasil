const _ = require('lodash');
const HD = require('@docbrasil/tools');
const Joi = require('joi');
const Boom = require('boom');
 
let internals = { };
 
internals.ui = {};
 
internals.ui.login = function (request, reply) {
    request.source = 'ui';
    HD.users().login(request, function(retAuth) {
        if(retAuth.auth) {
            let jwtOpts = HD.serverconfig().jwt;
            let stateOpts = { cookieKey: jwtOpts.cookieKey, id: request.user.sessionId, opts: jwtOpts.cookieOpts };
            HD.utils().server.respond(request, reply, null, retAuth, null, stateOpts);
 
        } else {
            HD.utils().server.respond(request, reply, Boom.unauthorized('Invalid user or passaword.'), retAuth);
        }
    });
}
 
internals.ui.logout = function (request, reply) {
    request.source = 'ui';
    HD.users().logout(request, function(retResp) {
        HD.utils().server.respond(request, reply, null, retResp);
    });
}
 
internals.admin = {};
 
internals.admin.login = function (request, reply) {
    request.source = 'admin';
    HD.users().login(request, function(retAuth) {
        if(retAuth.auth) {
            let jwtOpts = HD.serverconfig().jwt;
            let stateOpts = { cookieKey: jwtOpts.cookieKey, id: request.user.sessionId, opts: jwtOpts.cookieOpts };
            HD.utils().server.respond(request, reply, null, retAuth, null, stateOpts);
 
        } else {
            HD.utils().server.respond(request, reply, Boom.unauthorized('Invalid user or passaword.'), retAuth);
        }
    });
}
 
internals.admin.logout = function (request, reply) {
    request.source = 'admin';
    HD.users().logout(request, function(retResp) {
        HD.utils().server.respond(request, reply, null, retResp);
    });
}
 
module.exports = {
    routes: [
        {
            method: 'POST',
            path: '/login',
            config: {
                auth: { mode: 'optional' },
                handler: internals.ui.login,
                payload: { parse: true },
                validate: {
                    payload: {
                        username: Joi.string().required(),

                    }
                },
                description: 'Login the user',
